This document provides a complete implementation guide for
building a permissionless multi-node system where external
operators run AI model inference jobs on their laptops/servers,
earning cryptocurrency rewards through Polygon testnet smart
contracts. The system enables creators to upload models, buyers to
request inference jobs, and nodes to compete for computation tasks
in a decentralized marketplace.
Core Architecture: 3 dashboards (Creator, Buyer, Node) + Node
daemon software + Supabase backend + Polygon smart contracts +
TensorFlow Lite edge inference.

function _splitPayment(uint256 _jobId) internal {
Job storage job = jobs[_jobId];
uint256 nodeShare = (job.payment * 60) / 100;
uint256 creatorShare = (job.payment * 20) / 100;
uint256 protocolShare = job.payment - nodeShare - creatorShare;
payable(job.assignedNode).transfer(nodeShare);
// creatorShare sent to model creator (fetch from models mapping)
// protocolShare sent to protocol wallet
// Update node reputation
nodes[job.assignedNode].reputation += 10;
emit PaymentSplit(_jobId, nodeShare, creatorShare);
}

def poll_jobs():
"""Poll Supabase for pending jobs"""
headers = {"apikey": API_KEY, "Authorization": f"Bearer {API_KEY}"}
response = requests.get(
f"{BACKEND_URL}/rest/v1/jobs?
status=eq.pending&order=created_at.desc&limit=10",
headers=headers
)
return response.json()
def submit_bid(job_id, estimated_time):
"""Submit bid to backend"""
headers = {"apikey": API_KEY, "Content-Type": "application/json"}
bid_data = {
"job_id": job_id,
"node_id": NODE_ID,
"estimated_time_ms": estimated_time
}
response = requests.post(
f"{BACKEND_URL}/rest/v1/job_bids",
headers=headers,
json=bid_data
)
return response.status_code == 201
def execute_inference(job):
"""Run TFLite model on job input"""
# Download input data
input_data = requests.get(job["input_data_url"]).json()
# Load model
model_path = f"models/{job['model_id']}.tflite"
if not os.path.exists(model_path):
# Download from IPFS/S3
download_model(job["model_id"], model_path)
interpreter = tflite.Interpreter(model_path)
interpreter.allocate_tensors()
# Preprocess and infer
features = preprocess(input_data)
interpreter.set_tensor(0, features)
start = time.time()
interpreter.invoke()
latency = (time.time() - start) * 1000
output = interpreter.get_tensor(0)[0]
result = {"prediction": float(output), "latency_ms": latency}
# Compute hash
result_json = json.dumps(result, sort_keys=True)
result_hash = hashlib.sha256(result_json.encode()).hexdigest()
# Upload to IPFS
result_url = upload_to_ipfs(result_json)
return result_hash, result_url, latency
def submit_to_blockchain(job_id, result_hash, result_url):
"""Submit result to smart contract"""
contract = w3.eth.contract(address=CONTRACT_ADDRESS,
abi=CONTRACT_ABI)
txn = contract.functions.submitResult(
int(job_id),
Web3.toBytes(hexstr=result_hash),
result_url
).build_transaction({
'from': account.address,
'nonce': w3.eth.get_transaction_count(account.address),
'gas': 200000,
'gasPrice': w3.eth.gas_price
})
signed = w3.eth.account.sign_transaction(txn, WALLET_KEY)
tx_hash = w3.eth.send_raw_transaction(signed.rawTransaction)
receipt = w3.eth.wait_for_transaction_receipt(tx_hash)
return receipt.status == 1
